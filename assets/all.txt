.edit init
.tp 112 112
.pagebg black
.invis 1
.canmove 0
.exec mainmenu
.end

.edit mainmenu
.db new 56 56 96 56
.db write "               new game                load                    options"
.db mode menu
.db cursor 1 1
.db choice 0 0 1 1 newgame
.db cursor 1 3
.db choice 0 0 2 2 load
.db cursor 1 5
.db choice 1 1 2 2 options
.db cursor 1 1
.end

.edit newgame
.db delete
.invis 0
.canmove 1
.tp 269 272
.exec ents
.end

.edit ents
.ent create npc 36 300 300 16 16
.ent create interact 300 300 16 16 ".exec npcinteract"
.end

.edit dialogexit
.db delete
.canmove 1
.allow interact
.end

.edit npcdialog
.db new 80 140 100 64
.db write "             hello. i am an npc. i   do not have much to say                   ok"
.db mode menu
.db cursor 6 6
.db choice 0 0 0 0 dialogexit
.db cursor 6 6
.end

.edit npcinteract
.prevent interact
.canmove 0
.exec npcdialog
.end
